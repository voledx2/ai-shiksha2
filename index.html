# Scaled dot-product attention # AI Shiksha — Foundation Track, Lesson 4
def attention(Q, K, V, mask=None): d_k = Q.shape[-1] # Step 1: Compatibility scores scores = Q @ K.transpose(-2, -1) # Step 2: Scale (kyun? dot products # grow large → softmax saturates) scores = scores / d_k ** 0.5 if mask is not None: scores.masked_fill_(mask, -1e9) weights = F.softmax(scores, dim=-1) return weights @ V, weights
